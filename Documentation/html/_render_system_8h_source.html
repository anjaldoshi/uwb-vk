133  template<typename T>
134  void createUniformBuffer(std::shared_ptr<UBO>& ubo, const size_t& count)
135  {
136  std::cout << "sizeof: " << sizeof(T) << std::endl;
137  ubo = std::make_shared<UBO>();
138  size_t swapchainSize = mSwapchain->size();
139 
140  ubo->bufferSize = sizeof(T);
141 
142  ubo->buffers.resize(swapchainSize * count);
143  ubo->buffersMemory.resize(swapchainSize * count);
144  for (size_t i = 0; i < ubo->buffers.size(); i++) {
145  mBufferManager->createBuffer(ubo->bufferSize,
146  VK_BUFFER_USAGE_UNIFORM_BUFFER_BIT,
147  VK_MEMORY_PROPERTY_HOST_VISIBLE_BIT | VK_MEMORY_PROPERTY_HOST_COHERENT_BIT,
148  ubo->buffers[i],
149  ubo->buffersMemory[i]);
150  }
151 
152  mUniformBuffers.push_back(ubo);
153  }

163  template<typename T>
164  void updateUniformBuffer(const UBO& ubo, const T& uboData, size_t bufIndex)
165  {
166  //the actual index, taking into account duplicates for the swapchain
167  size_t actualIndex = (ubo.buffersMemory.size() / MAX_CONCURRENT_FRAMES) * mCurrentFrame + bufIndex;
168 
169  void* data;
170  vkMapMemory(mContext->device, ubo.buffersMemory[actualIndex], 0, sizeof(T), 0, &data);
171  memcpy(data, &uboData, sizeof(T));
172  vkUnmapMemory(mContext->device, ubo.buffersMemory[actualIndex]);
173  }
